var lay = new Vue({
  el: '#nest',
  props: ["recoloring"],
  data: {
    elements: {
      'box1': { value: 'first' },
      'box2': { value: '' },
      'box3': { value: '' },
      'box4': { value: '' }
    },
    elements_list: ['box1', 'box2', 'box3', 'box4'],
    colorInput: "",
  },
  computed:{
  	selectedBoxes(){
      return Object.values(this.elements)
        .filter(el => el.value)
        .map((el, ndx) => ({value: el.value, color: this.colors[ndx]}))
    },
    colors() {return this.colorInput.split(',')}
  },
});

      {{box.value}}
